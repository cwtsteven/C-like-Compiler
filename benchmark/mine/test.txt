int fibonacci(int n) {
	if (n <= 1) {
		return n;
	} else {
		return n + fibonacci(n - 1) + fibonacci(n - 2);
	}
}
int factorial(int n) {
	if (n <= 1) {
		return 1;
	} else {
		return n * factorial(n - 1);
	}
}
main() {
	int i = 1;
	int limit = 25;
	while (i <= limit) {
		int j = 1;
		while (j <= limit) {
			int k = 1;
			while (k <= limit) {
				if (i <= j + k) {
					>> fibonacci(i) + factorial(i);
				}
				k = k + 1;
			}
			j = j + 1;
		}
		i = i + 1;
	}
}
